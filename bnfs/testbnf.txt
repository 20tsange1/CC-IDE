<contract> ::= <start>+
;

<start> CC::= (<user> has <discount> if <negated>? <conditional>.)
;

<user> ::= <string>
;

<negated> ::= (it is not true that)
;

<discount> S::= <num>% (discount | off)?
;

<conditional> C::= <condition1> | <condition2> | <condition3>
;

<condition1> ::= (they are a member)
;

<condition2> ::= (they are male)
;

<condition3> ::= (they are female)
;

<string> ::= /[a-zA-Z]+/
 ;

<num> ::= /[0-9]+/
 ;
