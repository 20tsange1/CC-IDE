contract Freelance Agreement {
    parties {
        client: Alice Corp,
        freelancer: Bob Smith,
    }

    obligations {
        freelancer delivers Website Design by 2025-03-01,
        client pays freelancer $5000 upon delivery,
    }

    conditions {
        if work is delayed beyond 2025-03-10 then penalty is $100 per day,
    }

    termination {
        contract expires on 2025-12-31,
        if freelancer fails to deliver then contract is void,
    }

    signatures required from client, freelancer
}
